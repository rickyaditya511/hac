-- ==================== UPDATE OBJECT CACHE (DARI DEOBFUSCATOR) ====================
local function UpdateObjectCache()
    local map = workspace:FindFirstChild("Map")
    if not map then return end
    
    CachedObjects.Generators = {}
    CachedObjects.Pallets = {}
    CachedObjects.Hooks = {}
    CachedObjects.Gates = {}
    CachedObjects.Windows = {}
    CachedObjects.SCPs = {}
    
    for _, descendant in ipairs(map:GetDescendants()) do
        if descendant:IsA("Model") then
            local nameLower = string.lower(descendant.Name)
            if nameLower == "generator" then
                table.insert(CachedObjects.Generators, descendant)
            elseif nameLower == "hook" and descendant:FindFirstChild("HookPoint") then
                table.insert(CachedObjects.Hooks, descendant)
            elseif string.lower(descendant:GetAttribute("Type") or "") == "gate" then
                table.insert(CachedObjects.Gates, descendant)
            elseif string.lower(descendant:GetAttribute("Type") or "") == "window" then
                table.insert(CachedObjects.Windows, descendant)
            elseif string.match(nameLower, "^scp") then
                table.insert(CachedObjects.SCPs, descendant)
            end
        elseif descendant:IsA("BasePart") and descendant.Name == "PrimaryPartPallet" then
            local parent = descendant.Parent
            if parent and not table.find(CachedObjects.Pallets, parent) then
                table.insert(CachedObjects.Pallets, parent)
            end
        end
    end
    
    for _, descendant in ipairs(workspace:GetDescendants()) do
        if descendant:IsA("Model") and string.match(string.lower(descendant.Name), "^scp") then
            if not table.find(CachedObjects.SCPs, descendant) then
                table.insert(CachedObjects.SCPs, descendant)
            end
        end
    end
end

local function GetGeneratorProgress(generator)
    local progress = generator:GetAttribute("RepairProgress") or 0
    if type(progress) == "number" then
        if progress >= 0 and progress <= 1 then
            return math.floor(progress * 100)
        else
            return math.floor(progress)
        end
    end
    return 0
end

local function UpdatePlayerESP()
    if not ESPMaster then return end
    if not (PlayerESP or KillerESP) then return end
    
    local playerPos = nil
    if LocalPlayer.Character then
        local root = LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        if root then playerPos = root.Position end
    end
    
    for player, data in pairs(PlayerESPObjects) do
        local character = player.Character
        if character and character.Parent then
            local rootPart = character:FindFirstChild("HumanoidRootPart")
            if rootPart and playerPos then
                local dist = GetDistance(rootPart.Position, playerPos)
                local color = data.isKiller and ESPColors.Killer or ESPColors.Player
                
                if data.highlight then
                    data.highlight.FillColor = color
                    data.highlight.OutlineColor = color
                end
                
                if data.billboard and data.billboard.textLabel then
                    data.billboard.textLabel.Text = string.format("%s [%dm]", player.Name, math.floor(dist))
                    data.billboard.textLabel.TextColor3 = color
                    data.billboard.line.BackgroundColor3 = color
                end
                
                if data.isKiller and data.killerWarnBillboard and playerPos then
                    if dist <= 80 then
                        data.killerWarnBillboard.Visible = true
                        local warnLabel = data.killerWarnBillboard:FindFirstChild("WarnText")
                        if warnLabel then
                            if dist <= 40 then
                                warnLabel.Text = "‼️‼️‼️"
                                warnLabel.TextColor3 = Color3.fromRGB(255, 0, 0)
                            else
                                warnLabel.Text = "⚠️⚠️"
                                warnLabel.TextColor3 = Color3.fromRGB(255, 255, 0)
                            end
                        end
                    else
                        data.killerWarnBillboard.Visible = false
                    end
                end
                
                -- ESP Item Icon (dari deobfuscator)
                if ESPItemIcon and not data.isKiller then
                    local equippedItem = player:GetAttribute("EquippedItem")
                    if equippedItem and equippedItem ~= "" then
                        local icon = GetItemIcon(equippedItem)
                        if data.billboard and data.billboard.iconLabel then
                            data.billboard.iconLabel.Image = icon or ""
                            data.billboard.iconLabel.Visible = icon ~= nil
                        end
                    elseif data.billboard and data.billboard.iconLabel then
                        data.billboard.iconLabel.Visible = false
                    end
                end
            else
                RemovePlayerESP(player)
            end
        else
            RemovePlayerESP(player)
        end
    end
end

local function GetSilentTarget()
    local viewportCenter = CurrentCamera.ViewportSize / 2
    local closestDist = PistolFOVRadius
    local closestTarget = nil
    
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            if TargetSelection == "Killer" and not IsKiller(player) then goto continue end
            if TargetSelection == "Survivor" and not IsSurvivor(player) then goto continue end
            
            local targetPartObj = player.Character and player.Character:FindFirstChild(TargetPart)
            if targetPartObj then
                local targetPos = targetPartObj.Position
                if TargetPart == "Head" and SilentFlash then
                    targetPos = targetPartObj.Position + Vector3.new(0, FlashHeadOffset, 0)
                end
                local vec, onScreen = CurrentCamera:WorldToViewportPoint(targetPos)
                if onScreen then
                    local screenPos = Vector2.new(vec.X, vec.Y)
                    local centerDist = (screenPos - viewportCenter).Magnitude
                    
                    if centerDist < closestDist then
                        closestDist = centerDist
                        closestTarget = {player = player, part = targetPartObj, position = targetPos}
                    end
                end
            end
        end
        ::continue::
    end
    return closestTarget
end

-- ==================== AUTO PARRY (DARI DEOBFUSCATOR) ====================
local function CanParry()
    if ParryCooldown then return false end
    
    local character = LocalPlayer.Character
    if not character then return false end
    
    if character:GetAttribute("Knocked") or character:GetAttribute("IsHooked") then
        return false
    end
    
    if ParrySafety then
        local checkInterractable = character:FindFirstChild("CheckInterractable")
        if checkInterractable then
            if checkInterractable:GetAttribute("isVaulting") == true then return false end
            if checkInterractable:GetAttribute("isRepairing") == true then return false end
            if checkInterractable:GetAttribute("isUnhooking") == true then return false end
            if checkInterractable:GetAttribute("isHealing") == true then return false end
            if checkInterractable:GetAttribute("isSliding") == true then return false end
        end
    end
    
    return true
end

local function DoParry()
    if ParryCooldown then return end
    
    if Remotes.Items.Parry then
        for i = 1, 10 do
            Remotes.Items.Parry:FireServer()
        end
        
        if ParryCircle then

            if character and character:FindFirstChild("HumanoidRootPart") then

                task.delay(0.4, function() circle:Destroy() end)
            end
        end
    end
end

local function SetupAnimationListener()
    local character = LocalPlayer.Character
    if not character then return end
    
    local humanoid = character:FindFirstChild("Humanoid")
    if not humanoid then return end
    
    local animator = humanoid:FindFirstChildOfClass("Animator")
    if not animator then
        humanoid.ChildAdded:Connect(function(child)
            if child:IsA("Animator") then
                SetupAnimationListener()
            end
        end)
        return
    end
    
    if AutoParryConnection then AutoParryConnection:Disconnect() end
    
    AutoParryConnection = animator.AnimationPlayed:Connect(function(animationTrack)
        if not AutoParry then return end
        
        local animId = animationTrack.Animation and animationTrack.Animation.AnimationId or ""
        local animNum = animId:match("%d+")
        
        local lungeName = LUNGE_ANIMATIONS[animNum]
        if not lungeName then return end
        
        if IgnoredSkillsList and IgnoredSkillsList[lungeName] then return end
        
        if not CanParry() then return end
        
        local rootPart = character:FindFirstChild("HumanoidRootPart")
        if not rootPart then return end
        
        local nearestDist = ParryRadius
        local nearestAttackerPos = nil
        
        for _, player in ipairs(Players:GetPlayers()) do
            if player ~= LocalPlayer and IsKiller(player) and player.Character then
                local attackerRoot = player.Character:FindFirstChild("HumanoidRootPart")
                if attackerRoot then
                    local dist = GetDistance(attackerRoot.Position, rootPart.Position)
                    if dist < nearestDist then
                        nearestDist = dist
                        nearestAttackerPos = attackerRoot.Position
                    end
                end
            end
        end
        
        if nearestAttackerPos and nearestDist <= ParryRadius then
            if AggressiveParry then
                DoParry()
            else
                local playerPos = Vector3.new(rootPart.Position.X, 0, rootPart.Position.Z)
                local attackerPos = Vector3.new(nearestAttackerPos.X, 0, nearestAttackerPos.Z)
                local toAttacker = (attackerPos - playerPos).Unit
                local lookVector = Vector3.new(rootPart.CFrame.LookVector.X, 0, rootPart.CFrame.LookVector.Z).Unit
                
                if lookVector:Dot(toAttacker) >= ParryFace then
                    DoParry()
                end
            end
        end
    end)
end

                    if Remotes.Generator.SkillCheckResult then
                        Remotes.Generator.SkillCheckResult:FireServer(true)
                    end
                    PressSpace()
                    lastPress = tick()
                end
            end)
        end
        
        checkFrame:GetPropertyChangedSignal("Visible"):Connect(OnSkillCheckVisible)
        if checkFrame.Visible then OnSkillCheckVisible() end

-- ==================== AUTO GENERATOR (DARI DEOBFUSCATOR) ====================
local function GetNearestGenerator()
    local character = LocalPlayer.Character
    if not character then return nil end
    local rootPart = character:FindFirstChild("HumanoidRootPart")
    if not rootPart then return nil end
    
    local nearest = nil
    local nearestScore = math.huge
    
    for _, generator in ipairs(CachedObjects.Generators) do
        if generator and generator.Parent then
            local progress = GetGeneratorProgress(generator)
            if progress < 100 then
                local genPart = generator:FindFirstChildWhichIsA("BasePart")
                if genPart then
                    local dist = GetDistance(genPart.Position, rootPart.Position)
                    local priorityWeight = (100 - progress) * (GeneratorPriority / 100)
                    local score = dist - priorityWeight
                    if score < nearestScore then
                        nearestScore = score
                        nearest = generator
                    end
                end
            end
        end
    end
    return nearest
end

-- ==================== KILLER AIMBOT (DARI DEOBFUSCATOR) ====================
local function GetKillerTarget()
    local viewportCenter = CurrentCamera.ViewportSize / 2
    local closestDist = KillerFOVRadius
    local closestTarget = nil
    
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and IsSurvivor(player) and player.Character then
            local head = player.Character:FindFirstChild("Head")
            if head then
                local vec, onScreen = CurrentCamera:WorldToViewportPoint(head.Position)
                if onScreen then
                    local screenPos = Vector2.new(vec.X, vec.Y)
                    local centerDist = (screenPos - viewportCenter).Magnitude
                    local worldDist = GetDistance(head.Position, CurrentCamera.CFrame.Position)
                    
                    if centerDist < closestDist and worldDist <= KillerMaxDist then
                        closestDist = centerDist
                        closestTarget = head
                    end
                end
            end
        end
    end
    return closestTarget
end

        if input.UserInputType == Enum.UserInputType.MouseButton1 then dragging = false end
    end)
    speedSlider.InputChanged:Connect(function(input)
        if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then

                    if inputState == Enum.UserInputState.Change then
                        local moveDirection = Vector3.new()

                            moveDirection = moveDirection + CurrentCamera.CFrame.LookVector

                        if UserInputService:IsKeyDown(Enum.KeyCode.A) then
                            moveDirection = moveDirection - CurrentCamera.CFrame.RightVector
                        end
                        if UserInputService:IsKeyDown(Enum.KeyCode.D) then
                            moveDirection = moveDirection + CurrentCamera.CFrame.RightVector
                        end
                        if UserInputService:IsKeyDown(Enum.KeyCode.Space) then
                            moveDirection = moveDirection + Vector3.new(0, 1, 0)

-- ==================== AFK ESCAPE (DARI DEOBFUSCATOR) ====================
local function FindExitPosition()
    local map = workspace:FindFirstChild("Map")
    if not map then return Vector3.new(3098.16, 454.04, -4918.74) end
    
    if map:FindFirstChild("RooftopHitbox") or map:FindFirstChild("Rooftop") then
        return Vector3.new(3098.16, 454.04, -4918.74)
    elseif map:FindFirstChild("HooksMeat") then
        return Vector3.new(1546.12, 152.21, -796.72)
    elseif map:FindFirstChild("churchbell") then
        return Vector3.new(760.98, -20.14, -78.48)
    end
    
    local finishLine = map:FindFirstChild("Finishline")
    if finishLine then
        if finishLine:IsA("BasePart") then return finishLine.Position end
        local part = finishLine:FindFirstChildWhichIsA("BasePart")
        if part then return part.Position end
    end
    
    for _, descendant in ipairs(map:GetDescendants()) do
        if string.lower(descendant.Name):find("finish") then
            if descendant:IsA("BasePart") then return descendant.Position end
            local part = descendant:FindFirstChildWhichIsA("BasePart")
            if part then return part.Position end
        end
    end
    
    return Vector3.new(3098.16, 454.04, -4918.74)
end

-- ==================== CROSSHAIR (DARI DEOBFUSCATOR) ====================
local function UpdateCrosshair()
    for _, obj in pairs(CrosshairObjects) do
        if obj.Remove then obj:Remove() end
    end
    CrosshairObjects = {}
    
    if not CrosshairEnabled then return end
    
    local viewport = CurrentCamera.ViewportSize
    local centerX = viewport.X / 2 + CrosshairPosX
    local centerY = viewport.Y / 2
    
    if CrosshairStyle == "Dot" then
        local dot = Drawing.new("Circle")
        dot.Visible = true
        dot.Position = Vector2.new(centerX, centerY)
        dot.Radius = CrosshairSize
        dot.Color = CrosshairColor
        dot.Thickness = CrosshairThickness
        dot.Filled = true
        CrosshairObjects.dot = dot
        
    elseif CrosshairStyle == "Cross" then
        local hLine = Drawing.new("Line")
        hLine.Visible = true
        hLine.From = Vector2.new(centerX - CrosshairSize, centerY)
        hLine.To = Vector2.new(centerX + CrosshairSize, centerY)
        hLine.Color = CrosshairColor
        hLine.Thickness = CrosshairThickness
        CrosshairObjects.hLine = hLine
        
        local vLine = Drawing.new("Line")
        vLine.Visible = true
        vLine.From = Vector2.new(centerX, centerY - CrosshairSize)
        vLine.To = Vector2.new(centerX, centerY + CrosshairSize)
        vLine.Color = CrosshairColor
        vLine.Thickness = CrosshairThickness
        CrosshairObjects.vLine = vLine
        
    elseif CrosshairStyle == "Circle" then
        local circle = Drawing.new("Circle")
        circle.Visible = true
        circle.Position = Vector2.new(centerX, centerY)
        circle.Radius = CrosshairSize
        circle.Color = CrosshairColor
        circle.Thickness = CrosshairThickness
        circle.Filled = false
        CrosshairObjects.circle = circle
    end
end

-- ==================== NAME SPOOFER (DARI DEOBFUSCATOR) ====================
local SpooferConns = {}
local function SpoofUI()
    if not NameSpoofer then return end
    
    local playerGui = LocalPlayer:WaitForChild("PlayerGui")
    if not playerGui then return end
    
    local function SpoofDescendant(descendant)
        if descendant:IsA("TextLabel") or descendant:IsA("TextButton") or descendant:IsA("TextBox") then
            if not descendant:GetAttribute("OriginalText") then
                descendant:SetAttribute("OriginalText", descendant.Text)
            end
            local newText = descendant.Text
            for _, player in ipairs(Players:GetPlayers()) do
                newText = string.gsub(newText, player.Name, FakeName)
                newText = string.gsub(newText, player.DisplayName, FakeName)
            end
            descendant.Text = newText
        elseif descendant:IsA("ImageLabel") or descendant:IsA("ImageButton") then
            for _, player in ipairs(Players:GetPlayers()) do
                if string.find(descendant.Image, tostring(player.UserId)) then
                    if not descendant:GetAttribute("OriginalImage") then
                        descendant:SetAttribute("OriginalImage", descendant.Image)
                    end
                    descendant.Image = "rbxasset://textures/ui/GuiImagePlaceholder.png"
                end
            end
        end
    end
    
    for _, descendant in ipairs(playerGui:GetDescendants()) do
        SpoofDescendant(descendant)
    end
    
    local conn = playerGui.DescendantAdded:Connect(SpoofDescendant)
    table.insert(SpooferConns, conn)
end

local function UnspoofUI()
    for _, conn in ipairs(SpooferConns) do
        conn:Disconnect()
    end
    SpooferConns = {}
    
    local playerGui = LocalPlayer:FindFirstChild("PlayerGui")
    if playerGui then
        for _, descendant in ipairs(playerGui:GetDescendants()) do
            if descendant:IsA("TextLabel") or descendant:IsA("TextButton") or descendant:IsA("TextBox") then
                local original = descendant:GetAttribute("OriginalText")
                if original then
                    descendant.Text = original
                    descendant:SetAttribute("OriginalText", nil)
                end
            elseif descendant:IsA("ImageLabel") or descendant:IsA("ImageButton") then
                local original = descendant:GetAttribute("OriginalImage")
                if original then
                    descendant.Image = original
                    descendant:SetAttribute("OriginalImage", nil)
                end
            end
        end
    end
end
